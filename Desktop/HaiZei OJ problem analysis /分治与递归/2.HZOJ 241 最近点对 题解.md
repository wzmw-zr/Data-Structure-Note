# HZOJ 241 最近点对 题解

## 题目描述

 有 A,B 两组点，每组点的数量为 N，从 A 中选一个点，再从 B 中选一个点，求这两个点之间的最短距离是多少。



## 分析

最近点对类型的问题就是分治与递归解法，这里依旧需要对点对进行排序，并且需要标明点所属的集合。



## 代码

```c++
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
#define MAX_N 100000

struct Node {
    double x, y;
    int z;
};

int n;
Node p[2 * MAX_N + 5];

int cmp(Node a, Node b) {
    if (a.x == b.x) return a.y < b.y;
    return a.x < b.x;
}

double dis(int l, int r) {
    return sqrt(pow(p[l].x - p[r].x, 2) + pow(p[l].y - p[r].y, 2));
}

double func(int l, int r) {
    if (l == r) return 2100000000;
    if (l + 1 == r) {
        // 注意两个点在不同集合与在相同集合中的情况
        if (p[l].z != p[r].z) return dis(l, r);
        return 2100000000;
    }
    int mid = (l + r) / 2;
    double d = min(func(l, mid), func(mid + 1, r));
    // i， j的范围需要注意，边界都是可以取到的。
    for (int i = mid; i >= l && p[mid].x - p[i].x < d; i--) {
        for (int j = mid + 1; j <= r && p[j].x - p[mid].x < d; j++) {
            // 在不同集合中才更新
            if (p[i].z != p[j].z) d = min(d, dis(i, j));
        }
    }
    return d;
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%lf%lf", &p[i].x, &p[i].y);
        p[i].z = 1;
    }
    for (int i = 2 * n - 1; i >= n; i--) {
        scanf("%lf%lf", &p[i].x, &p[i].y);
        p[i].z = 0;
    }
    sort(p, p + 2 * n, cmp);
    double ans = func(0, 2 * n - 1);
    printf("%.3lf\n", ans);
    return 0;
}

```
